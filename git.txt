	git rm --cached ---.---		-> namefile
	git commit -am ''
	git commit -m "change file - resolve #1"
	git commit --amend

	git show 1234567 		-> The first seven digits of the IDs that were committed
	git show 1234567 -- ---.--- 		-> It is possible that several files have been changed in one commit, but this method shows the changes of the desired file in one commit & namefile
